cmake_minimum_required(VERSION 3.12)
project(srt_python_module)

# Указываем стандарт C++
set(CMAKE_CXX_STANDARD 11)

# Находим pybind11
find_package(pybind11 REQUIRED)

# Находим библиотеку SRT (необходимо правильно указать путь к библиотекам и заголовкам SRT)
find_library(SRT_LIBRARY NAMES srt)
include_directories(/root/Python-3.7.17/str_module/includes)

# Создаем модуль для Python
pybind11_add_module(srt_libsrt srt_libsrt.cpp)

# Линкуем библиотеку SRT
target_link_libraries(srt_libsrt PRIVATE ${SRT_LIBRARY})
